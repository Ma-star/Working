\subsection{桥接网络配置}
    \begin{lstlisting}
        auto br-ex
        iface br-ex inet static
        address 172.16.19.147
        netmask 255.255.255.0
        gateway 172.16.19.254
        dns-nameservers 172.16.5.1

        auto eth0
        iface eth0 inet manual
        up ifconfig $IFACE 0.0.0.0 up
        up ip link set $IFACE promisc on
        down ip link set $IFACE promisc off
        down ifconfig $IFACE down
    \end{lstlisting}
\subsection{临时更换mac地址}
    \begin{itemize}
        \item[1.]ovs-vsctl del-br br-ex 删除br-ex网桥
        \item[2.]ovs-vsctl add-br br-ex 添加br-ex网桥
        \item[3.]ip link list 查看br-ex和eth0的mac地址
        \item[4.]ip link set br-ex addr eth0的mac地址(由于网络安全限制，ip和mac需绑定，不是很明白)
        \item[5.]ovs-vsctl add-port br-ex eth0 将eth0桥接到br-ex虚拟网桥
    \end{itemize}
