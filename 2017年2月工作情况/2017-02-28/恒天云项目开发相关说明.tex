\section{恒天云项目开发相关说明}

\subsection{说明一：从恒天云的git获取下载代码到自己用户所属的开发环境}
\begin{itemize}
	\item[1.] 首先通过ssh命令，远程到开发环境的机器
        \begin{lstlisting}
            ssh weiguangzhou@172.16.7.1
        \end{lstlisting}
        
	\item[2.] 从172.16.5.77的git服务器进行克隆
        \begin{lstlisting}
            git clone git@172.16.5.77:hengtianyun/hengtianyun.git
        \end{lstlisting}
	\item[3.] 查看git仓库的所有分支，默认第一个是master
        \begin{lstlisting}
            git branch -a
        \end{lstlisting}
	\item[4.] 通过pull命令，再次更新代码
        \begin{lstlisting}
            git pull
        \end{lstlisting}
	\item[5.] 获取horizon的分支
        \begin{lstlisting}
            git checkout hengtianyun_v3.9_dev
        \end{lstlisting}
\end{itemize}

\subsection{说明二：使用pycharm调试horizon}
\begin{itemize}
	\item[1.] 进入coding,选择一个默认项目，打开，点击右边菜单栏的wiki
	\item[2.] 在wiki中找到pycharm调试horizon的文档，进行配置
\end{itemize}
